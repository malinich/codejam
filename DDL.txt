CREATE COLUMN TABLE "USER"."STOPS"(
	"STOP_ID" INTEGER CS_INT,
	"STOP_NAME" NVARCHAR(100),
	"STOP_LAT" DOUBLE CS_DOUBLE,
	"STOP_LON" DOUBLE CS_DOUBLE,
	PRIMARY KEY (
		"STOP_ID"
	)
) UNLOAD PRIORITY 5 AUTO MERGE;
CREATE UNIQUE INDEX "UK" ON "USER"."STOPS" ( "STOP_NAME" ASC )
;

CREATE COLUMN TABLE "USER"."ROUTES"(
	"ROUTE_ID" INTEGER CS_INT,
	"ROUTE_TYPE" INTEGER CS_INT,
	"ROUTE_DESC" NVARCHAR(100),
	PRIMARY KEY (
		"ROUTE_ID"
	)
) UNLOAD PRIORITY 5 AUTO MERGE;

CREATE COLUMN TABLE "USER"."STOP_TIMES"(
	"TRIP_ID" INTEGER CS_INT,
	"ARRIVAL_TIME" TIME CS_SECONDTIME,
	"DEPARTURE_TIME" TIME CS_SECONDTIME,
	"STOP_ID" INTEGER CS_INT,
	"STOP_SEQUENCE" INTEGER CS_INT
) UNLOAD PRIORITY 5 AUTO MERGE;


CREATE COLUMN TABLE "USER"."TRIPS"(
	"ROUTE_ID" INTEGER CS_INT,
	"TRIP_ID" INTEGER CS_INT,
	"DIRECTION_ID" INTEGER CS_INT
) UNLOAD PRIORITY 5 AUTO MERGE;
